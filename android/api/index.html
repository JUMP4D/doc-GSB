    Documentation de l'API de GSB VISITES

    Principales technologies utilisées

    • TypeScript : Langage de programmation utilisé pour le développement backend.
    • Express : Framework web basé sur Node.js pour la création d'APIs REST.
    • MongoDB : Base de données NoSQL pour stocker les données des utilisateurs, des visites, des motifs et des praticiens.
    • dotenv : Bibliothèque pour la gestion des variables d'environnement (clés API, etc.).
    • express-validator : Utilisé pour valider les données saisies par l'utilisateur.
    • express-rate-limit : Utilisé pour limiter le nombre de requêtes effectuées dans un délai donné, protégeant ainsi l'API contre les attaques par brute force ou les attaques par déni de service (DoS).
    • bcrypt : Utilisé pour le hachage des mots de passe, garantissant qu'ils ne sont jamais stockés en clair.
    • jsonwebtoken : Bibliothèque utilisée pour générer et vérifier les tokens JWT, permettant une gestion sécurisée de l'authentification.

    Exemple de model :

    import mongoose from 'mongoose';
import encrypt from 'mongoose-encryption';
import bcrypt from 'bcrypt';
import dotenv from 'dotenv';

dotenv.config();

interface IVisiteur {
  nom: string;
  prenom: string;
  tel: string;
  email: string;
  date_embauche: Date;
  visite: mongoose.Schema.Types.ObjectId[];
  password: string;
}

const visiteurSchema = new mongoose.Schema<IVisiteur>({
    nom: { type: String, required: true},
    prenom: { type: String, required: true },
    tel: { type: String, required: true },
    email: { type: String, required: true, unique: true },
    date_embauche: { type: Date, required: true },
    visite: [{
        type: mongoose.Schema.Types.ObjectId,
        ref: 'Visite',
        required: true
    }],
    password: { type: String, required: true }
  });

visiteurSchema.plugin(encrypt, 
  { secret: process.env.KEY,
    encryptedFields: ['nom', 'prenom', 'tel', 'date_embauche'] 
});

visiteurSchema.pre('save', async function (next) {
  if (this.isModified('password')) {
    this.password = await bcrypt.hash(this.password, 1);
  }
  next();
});

// Méthode pour vérifier le mot de passe
visiteurSchema.methods.comparePassword = async function (candidatePassword: string) {
  return await bcrypt.compare(candidatePassword, this.password);
};

const Visiteur = mongoose.model<IVisiteur>('Visiteur', visiteurSchema);

export default Visiteur;

    Exemple de méthode du controlleur :

    export const createVisiteur = async (req: Request, res: Response): Promise<void> => {
  await body("email").isEmail().withMessage("Veuillez fournir un email valide.").run(req);
  await body("password")
    .isLength({ min: 6 })
    .withMessage("Le mot de passe doit contenir au moins 6 caractères.")
    .matches(/\d/).withMessage('Le mot de passe doit contenir au moins un chiffre.')
    .run(req);

  // Vérification des erreurs
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    res.status(400).json({ message: "Erreur de validation", errors: errors.array() });
    return;
  }

  try {
    const { nom, prenom, email, tel , date_embauche, password} = req.body;

    if (!nom || !prenom || !email || !tel || !date_embauche || !password) {
      res.status(400).json({ message: "Veuillez fournir un nom, un prénom, un email et un numéro de téléphone." });
      return;
    }

    // Vérification des types
    if (typeof nom !== 'string' || typeof prenom !== 'string' || typeof email !== 'string' || typeof tel !== 'string' || typeof date_embauche !== 'string' || typeof password !== 'string') {
      res.status(400).json({ message: "Tous les champs doivent être de type string." });
      return;
    }

    const newVisiteur = new Visiteur({ nom, prenom, email, tel, date_embauche, password });

    await newVisiteur.save();

    res.status(201).json(newVisiteur);
  } catch (err) {
    console.error("Erreur lors de la création du visiteur :", err);
    res.status(500).json({ message: "Erreur lors de la création du visiteur.", error: err instanceof Error ? err.message : err });
  }
}

    Exemple de route :

    import express from 'express';
import { getAllVisiteurs, createVisiteur, getVisiteurById, deleteVisiteur } from "../controllers/visiteurController";

const router = express.Router();

// Route pour lister tous les visiteurs (GET /visiteur)
router.get('/', getAllVisiteurs);

// Route pour créer un nouveau visiteur (POST /visiteur)
router.post('/', createVisiteur);

// Route pour récupérer un visiteur par son ID (GET /visiteur/:id)
router.get('/:id', getVisiteurById);

// Route pour supprimer un visiteur par son ID (DELETE /visiteur/:id)
router.delete('/:id', deleteVisiteur);

export default router;
